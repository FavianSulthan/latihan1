game_start = input("Do you want to play a game?(yes/no)").lower()
if game_start == "no":
    print("Alright, Have a nice day!")
elif game_start == "yes":
    name = input("What is your name?")
    age = int(input("What is your age?"))
    health = 1
    if age <= 12:
        print("Your age doesn\'t meet the requirements")
    else:
        print("You are old enough to play!")
        start = input("Start playing?(yes/no)").lower()
        if start == "yes":
            print("You are starting with", health,"health")
            print("Let's Start!")
            awal = input('You are walking in a forest, you see a man sitting near a tree, '
                         'you walk towards him and he asks you for money to eat. '
                         'Will you give him some money?(yes/no)').lower()
            if awal == "yes":
                print("Oh no, he turned out to be a burglar. He took all of your money and killed you")
                health = 0
                print("You now have", health, "health")
                if health < 1:
                    print("You Lose")
            elif awal == "no":
                print("You are safe, that man was a burglar and wanted to steal your money.")
            if health > 0:
                tengah = input("You see a huge river in front of you, but there is a bridge to get pass through. "
                                   "would you walk on the bridge?(yes/no)").lower()
                if tengah == "yes":
                    print("While you walk on the bridge, the bridge breaks because it was already rusty and damaged, "
                      "you fall to the river.")
                    health = 0
                    print("You now have", health, "health")
                if health < 1:
                    print("You Lose")
                else:
                    print("The bridge looks rusty and very dangerous to walk on. "
                      "Let\'s find a new solution to get pass the river.")
            if health > 0:
                terakhir = input("There is a wooden boat. The boat is pretty small, "
                                 "while the river flow is very strong. It looks pretty dangerous. "
                                 "Would you ride the boat?(yes/no)").lower()
                if terakhir == "no":
                    print("You just missed the opportunity to pass the river. You now have no way to go back home.")
                    health = 0
                    print("You now have", health, "health")
                if health < 1:
                    print("You Lose")
                else:
                    print("Miraculously you passed the river. "
                      "You got to the other side of the forest. You now can go back to your house.")
                if health == 1:
                    print("You won the game! Congratulations.")
